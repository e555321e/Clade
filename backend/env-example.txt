# ========== AI 模型配置 ==========
# AI 服务商类型 (openai / deepseek / anthropic / custom)
AI_PROVIDER=openai

# API 端点 (根据服务商填写)
# DeepSeek 官方: https://api.deepseek.com
# 硅基流动: https://api.siliconflow.cn/v1
# OpenAI 官方: https://api.openai.com/v1
AI_BASE_URL=https://api.siliconflow.cn/v1

# API 密钥 (必填)
AI_API_KEY=sk-your-api-key-here

# 默认模型名称
# DeepSeek 官方: deepseek-chat 或 deepseek-reasoner
# 硅基流动: deepseek-ai/DeepSeek-V3.2-Exp (免费) 或 Pro/deepseek-ai/DeepSeek-V3.2-Exp (付费)
AI_MODEL=deepseek-ai/DeepSeek-V3.2-Exp

# AI 请求超时时间 (秒)
AI_TIMEOUT=60

# AI 并发请求限制
AI_CONCURRENCY_LIMIT=15

# ========== Embedding 向量服务配置 ==========
# Embedding 服务商类型
EMBEDDING_PROVIDER=openai

# Embedding API 端点
EMBEDDING_BASE_URL=https://api.siliconflow.cn/v1

# Embedding API 密钥 (如果与 AI 相同可省略)
EMBEDDING_API_KEY=sk-your-api-key-here

# Embedding 模型名称
EMBEDDING_MODEL=Qwen/Qwen3-Embedding-4B

# ========== 日志配置 ==========
# 日志级别 (DEBUG / INFO / WARNING / ERROR)

# 是否输出日志到文件
LOG_TO_FILE=true

# 是否输出日志到控制台
